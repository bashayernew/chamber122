-- Fix RLS policies for events table -- Allow public read access to published events CREATE POLICY "Allow public read access to published events" ON events FOR SELECT USING ( status = 'published' AND is_published = true ); -- Allow authenticated users to insert events CREATE POLICY "Allow authenticated users to insert events" ON events FOR INSERT WITH CHECK ( auth.role() = 'authenticated' ); -- Allow users to update their own events CREATE POLICY "Allow users to update their own events" ON events FOR UPDATE USING ( auth.role() = 'authenticated' AND business_id IN ( SELECT id FROM businesses WHERE owner_id = auth.uid() ) ); -- Allow users to delete their own events CREATE POLICY "Allow users to delete their own events" ON events FOR DELETE USING ( auth.role() = 'authenticated' AND business_id IN ( SELECT id FROM businesses WHERE owner_id = auth.uid() ) );

Step 2: Alternative - Disable RLS Temporarily

If the above doesn't work, temporarily disable RLS on events table:

-- TEMPORARY: Disable RLS on events table ALTER TABLE events DISABLE ROW LEVEL SECURITY;
